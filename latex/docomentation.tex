\section{Project Overview}
%%============================================================================

This document describes a \textbf{production-grade, containerised full-stack
application} consisting of four services orchestrated via a single
\texttt{docker-compose.yml}. The mono-repo contains the backend, frontend, Nginx
configuration, and the complete CI/CD pipeline definition.

\begin{greenbox}
Every \texttt{git push origin main} triggers a fully automated build, push,
and deploy cycle. \textbf{Zero manual intervention is required} after the
initial one-time VM setup.
\end{greenbox}

\subsection{Project Structure}

\begin{lstlisting}[language=bash, caption={Mono-repo directory layout}]
/root
  docker-compose.yml         # 4-service orchestration
  .env.example               # Environment variable template
  .gitignore

  backend/
    index.js                 # Express server entry point
    Dockerfile
    config/db.js             # Connection pool
    migrations/              # node-pg-migrate files
    routes/user.routes.js

  frontend/
    src/                     # React source
    vite.config.js
    Dockerfile

  nginx/
    nginx.conf               # Reverse proxy rules
    Dockerfile

  .github/
    workflows/deploy.yml     # GitHub Actions CI/CD pipeline
\end{lstlisting}

\subsection{Service Details}

\subsubsection{PostgreSQL (Database)}
\begin{itemize}[leftmargin=1.6em, itemsep=3pt]
  \item \textbf{Port:} 5432 — internal only, never exposed externally
  \item \textbf{Storage:} Persistent volume \texttt{pgdata:/var/lib/postgresql/data}
  \item \textbf{Memory tuning:} \texttt{shared\_buffers=64MB},
        \texttt{max\_connections=50}
  \item \textbf{Health check:} \texttt{pg\_isready -U \$POSTGRES\_USER}
        every 10 seconds
\end{itemize}

\subsubsection{Backend (Express API)}
\begin{itemize}[leftmargin=1.6em, itemsep=3pt]
  \item \textbf{Port:} 5000 — internal only
  \item \textbf{Startup command:} \texttt{npm run migrate \&\& node index.js}
  \item \textbf{Responsibilities:} run DB migrations on every start,
        serve REST endpoints, authentication, connection-pool query execution
\end{itemize}

\subsubsection{Frontend (React SPA)}
\begin{itemize}[leftmargin=1.6em, itemsep=3pt]
  \item \textbf{Port:} 3000 — internal only
  \item \textbf{Runtime:} Node.js 22 Alpine, served by the \texttt{serve} package
  \item \textbf{Build:} Vite production build baked into the Docker image
\end{itemize}

\subsubsection{Nginx (Reverse Proxy)}
\begin{itemize}[leftmargin=1.6em, itemsep=3pt]
  \item \textbf{Port:} 80 — the \emph{only} publicly exposed port
  \item \textbf{Routes:}
        \texttt{/} $\to$ frontend:3000\quad
        \texttt{/api/} $\to$ backend:5000\quad
        \texttt{/health} $\to$ backend:5000
  \item \textbf{DNS resolver:} \texttt{127.0.0.11:53} (Docker internal DNS)
        for dynamic hostname resolution
\end{itemize}

\subsection{Rollback Strategy}

\begin{notebox}
If a deployment breaks production, previous images remain in OCIR.
SSH to the VM, edit \texttt{.env} to set \texttt{TAG=previous-working-tag},
then re-run \texttt{docker compose pull \&\& docker compose up -d}.
Alternatively, revert the commit on \texttt{main} and push again — the pipeline
will rebuild and redeploy automatically.
\end{notebox}

\subsection{Nginx DNS Resolution}

Docker's internal DNS resolver at \texttt{127.0.0.11} allows Nginx to resolve
container hostnames dynamically. Without this, Nginx would cache the IP at
startup and fail if a container restarts with a new internal IP.

\begin{lstlisting}[language=bash,
  caption={Relevant extract from \texttt{nginx/nginx.conf}}]
resolver 127.0.0.11 valid=10s;
resolver_timeout 5s;

location / {
    proxy_pass http://frontend:3000;  # resolved via Docker DNS
}
location /api/ {
    proxy_pass http://backend:5000;
}
location /health {
    proxy_pass http://backend:5000/health;
}
\end{lstlisting}

\subsection{Pre-Deployment Checklist}

\begin{itemize}[leftmargin=1.8em, itemsep=5pt, topsep=4pt]
  \item[\cmark] All code committed and pushed to the \texttt{main} branch
  \item[\cmark] All 8 GitHub Actions Secrets configured

  \item[\cmark] OCI Security List: TCP Ingress rule added for port~80
  \item[\cmark] SSH private key registered on VM and matching GitHub Secret
  \item[\cmark] OCIR credentials verified in VM \texttt{.env}
\end{itemize}

\end{lstlisting}

\end{lstlisting}

\begin{longtable}{>{\bfseries\small}p{3.8cm}
                  >{\small}p{3.8cm}
                  >{\small}p{5.6cm}}
  \toprule
  \rowcolor{paperMid} Issue & Likely Cause & Resolution \\
  \midrule
  \endfirsthead

  \endhead
  \bottomrule
  \endlastfoot

  Push image fails &
    OCIR credentials invalid &
    Verify Secrets: \texttt{OCIR\_USERNAME}, \texttt{OCIR\_TOKEN},
    \texttt{OCIR\_REGISTRY} \\[3pt]
  \rowcolor{paperStripe}
  SSH connection fails &
    Key format wrong &
    Must be PEM format; run
    \texttt{ssh-keygen -p -N "" -m pem -f id\_rsa} \\[3pt]

  App won't start on VM &
    Images not pulled &
    \texttt{docker compose pull -{}-no-parallel}
    then \texttt{docker compose up -d} \\[3pt]
  \rowcolor{paperStripe}
  Migrations fail &
    Syntax error in file &
    Check \texttt{migrations/} files against node-pg-migrate docs \\[3pt]

  Backend can't reach DB &
    Wrong \texttt{DATABASE\_URL} &
    Format must be:
    \texttt{postgres://user:pass@postgres:5432/db} \\[3pt]
  \rowcolor{paperStripe}
  Frontend returns 404 &
    Nginx config wrong &
    Ensure \texttt{location /} proxies to
    \texttt{http://frontend:3000} \\[3pt]

  Port 80 unreachable &
    OCI firewall &
    OCI Console $\to$ VCN $\to$ Security List
    $\to$ add TCP Ingress for port~80 \\[3pt]
  \rowcolor{paperStripe}
  Disk full on VM &
    Stale images &
    \texttt{docker image prune -a \&\&
    docker container prune} \\[3pt]

  Containers restarting &
    App crash on start &
    \texttt{docker compose logs \{service\}}
    to inspect the error \\[3pt]
  \rowcolor{paperStripe}
  High CPU usage &
    Connection pool exhausted &
    Review \texttt{max\_connections=50} in the postgres service config \\

\end{longtable}
